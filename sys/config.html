# 模块配置

# 权限接口模块配置参数

#是否验证AppId
xci.api.valid-app-id=true

#是否验证时间戳
xci.api.valid-timestamp=false

#验证时间戳时验证偏差(单位秒)
xci.api.timestamp-offset=120

#未授权错误消息
xci.api.un-authorized-msg=您没有权限请联系管理员

#权限Api接口模块配置拦截器忽略拦截的Url模式
xci.api.exclude-intercept-url-patterns=/favicon.ico, /doc.html, /db.html, /gen.html, /webjars/**, /swagger-resources

# 权限WebUI模块配置参数

#登录成功后默认Url
xci.web.default-url=/sys

#登录Url
xci.web.login-url=/sys/login

#HomeUrl
xci.web.home-url=/sys/home

#未授权Url
xci.web.un-authorized-url="/sys/unauthorized"

#登录超时错误消息
xci.web.login-timeout-msg=登录超时请重新登录

#未授权错误消息
xci.web.un-authorized-msg=您没有权限请联系管理员

#权限UI模块配置拦截器拦截Url模式
xci.web.intercept-url-patterns=/**

#权限UI模块模块配置拦截器忽略拦截的Url模式
xci.web.exclude-intercept-url-patterns=/api/**,/**/js/**,/**/css/**,/**/font/**,/**/img/**,/**/*.js,/**/*.css,/**/*.jpg,/**/*.png,/favicon.ico,/doc.html,/db.html,/gen.html,/webjars/**,/swagger-resources
