# SysWebService

SysWebService 
@Service
public class SysWebService {
    /**
     * 模块列表转为树节点列表
     * @param moduleList   模块列表
     * @param eachCallback 每行循环的回调函数
     */
    public List<WebTreeNode> toModuleNodeList(List<SysModule> moduleList, Consumer<WebTreeNode> eachCallback);

    /**
     * 机构列表转为树节点列表
     * @param deptList     机构列表
     * @param eachCallback 每行循环的回调函数
     */
    public List<WebTreeNode> toDeptNodeList(List<SysDept> deptList, Consumer<WebTreeNode> eachCallback);

    /**
     * 字典列表转为树节点列表
     * @param dicList 字典列表
     */
    public List<WebTreeNode> toDicNodeList(List<SysDic> dicList);

    /**
     * 字典类型列表转为树节点列表
     * @param dicCategoryList 字典类型列表
     */
    public List<WebTreeNode> toDicCategoryNodeList(List<SysDicCategory> dicCategoryList);
}
