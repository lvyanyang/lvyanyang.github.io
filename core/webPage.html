# 页面全局变量

系统在所有Controller对象的方法中全局注入WebPage类的实例，在所有ftl页面中统一使用web来调用。

加载 xci-framework-sys 模块后，还会全局注入SysWebPage对象，在所有ftl页面中统一使用sysWeb来调用。

# FreeMarker全局辅助文件

新建辅助文件路径：resources\freemarker_implicit.ftl，此文件的作用是给Idea开发工具提供类型说明，实现编辑器的代码提示。

[#ftl]
[#-- @implicitly included --]
[#-- @ftlroot "../resources/templates/" --]
[#-- @ftlroot "../resources/static/" --]
[#-- @ftlvariable name="_create_" type="java.lang.String" --]
[#-- @ftlvariable name="request" type="org.springframework.web.servlet.support.RequestContext" --]
[#-- @ftlvariable name="web" type="com.github.lvyanyang.component.WebPage" --]

# 使用方法

定义一个变量defOperateEndDateTime，值=当前时间
<#assign defOperateEndDateTime=web.formatDate(web.now())/>

输出客户端ip
${web.clientIP()}

生成一个cdn链接的css资源
<link href="${web.cdn('/jx.css')}" rel="stylesheet"/>

判断是否为空，控制输出
<#if web.isNotBlank(entity.afterData)>
    <tr>
        <th>操作后数据</th>
        <td>
            <pre>${entity.afterData}</pre>
        </td>
    </tr>
</#if>

时间格式化输出
${web.formatDateTime(entity.operateDateTime)}

# 全局注入自定义对象

  1. 定义@Component类
查看代码 
@Component
public class SysWebPage {
    @Resource private SysUserService userService;

    /**
     * 用户权限编码鉴权
     * @param code 权限编码
     * @return 拥有返回true
     */
    public boolean auth(String code) {
        return userService.authorize(userService.getCurrentUserId(), code);
    }

    /**
     * 获取当前登录用户
     */
    public SysUser currentUser() {
        return userService.getCurrentUser();
    }
}

  1. 定义ControllerAdvice类

/**
 * 全局注入变量到Web页面中
 */
@ControllerAdvice
public class SysWebControllerAdvice {
    //上面定义的组件类
    @Resource private SysWebPage sysWebPage;

    //注解参数就是页面引用的名称
    @ModelAttribute("sysWeb")
    public SysWebPage initSysWebPage() {
        return this.sysWebPage;
    }
}

@Component
public class WebPage {
    /**
     * 判断当前页面是否新建记录
     */
    public boolean isCreate();

    /**
     * 获取当前日期时间对象
     */
    public Date now();

    /**
     * 获取当前日期时间对象
     * @param offsetDay 偏移天数，正数为向后偏移，负数为向前偏移
     */
    public Date now(int offsetDay);

    /**
     * 获取指定日期偏移指定时间后的时间，生成的偏移日期不影响原日期
     * @param date      基准日期
     * @param offsetDay 偏移天数，正数为向后偏移，负数为向前偏移
     * @return 偏移后的日期
     */
    public static DateTime dateOffset(Date date, int offsetDay);

    /**
     * 格式化日期(默认格式yyyy-MM-dd)
     * @param date 指定的时间
     * @return 返回日期格式化后的字符串
     */
    public String formatDate(Date date);

    /**
     * 格式化日期字符串(默认格式 yyyy-MM-dd HH:mm)
     * @param date 指定的时间
     * @return 返回日期格式化后的字符串
     */
    public String formatDateHasMinute(Date date);

    /**
     * 格式化日期(默认格式 yyyy-MM-dd HH:mm:ss)
     * @param date 指定的时间
     * @return 返回日期格式化后的字符串
     */
    public String formatDateTime(Date date);

    /**
     * 获取格式化的时间字符串(默认格式 HH:mm:ss)
     * @param date 指定的时间
     * @return 返回日期格式化后的字符串
     */
    public String formatTime(Date date);

    /**
     * 格式化日期
     * @param date   指定的时间
     * @param format 格式字符串
     * @return 返回日期格式化后的字符串
     */
    public String formatDate(Date date, String format) ;

    /**
     * 以 conjunction 为分隔符将多个对象连接为字符串
     * @param conjunction 分隔符
     * @param objs        数组
     * @return 连接后的字符串
     * @see ArrayUtil#join(Object, CharSequence)
     */
    public String join(CharSequence conjunction, Object... objs);

    /**
     * 切分字符串
     * @param str       被切分的字符串
     * @param separator 分隔符字符
     * @return 切分后的数组
     */
    public static String[] splitToArray(CharSequence str, char separator);

    /**
     * 判读字符串是否包含另一个子串
     * @param source 目标
     * @param sub    子串
     */
    public boolean contains(String source, String sub);

    /**
     * 字符串是否为空白 空白的定义如下： <br>
     * 1、为null <br>
     * 2、为不可见字符（如空格）<br>
     * 3、""<br>
     * @param str 被检测的字符串
     * @return 是否为空
     */
    public boolean isBlank(CharSequence str);

    /**
     * 字符串是否为非空白 空白的定义如下： <br>
     * 1、不为null <br>
     * 2、不为不可见字符（如空格）<br>
     * 3、不为""<br>
     * @param str 被检测的字符串
     * @return 是否为非空
     */
    public boolean isNotBlank(CharSequence str);

    /**
     * 判断对象是否为null
     * @param obj 测试对象
     */
    public boolean isNull(Object obj);

    /**
     * 判断对象是否不为null
     * @param obj 测试对象
     */
    public boolean isNotNull(Object obj);

    /**
     * 获取Web项目下的web root路径
     * @return web root路径
     * @since 4.0.13
     */
    public File webRoot();

    /**
     * 获取站点完整Url信息
     */
    public String siteFullUrl();

    /**
     * 获取站点完整Url信息
     */
    public String siteFullUrl(String virtualUrl);

    /**
     * 获得文件的扩展名，扩展名不带“.”
     * @param fileName 文件名
     * @return 扩展名
     */
    public String extName(String fileName);

    /**
     * 返回主文件名
     * @param fileName 完整文件名
     * @return 主文件名
     */
    public String mainName(String fileName);

    /**
     * json序列化
     * @param data 待序列化的对象
     */
    public String toJsonString(Object data);

    /**
     * 获取客户端IP
     * @return IP地址
     */
    public String clientIP();

    /**
     * 获取 request Attribute参数
     * @param name 属性名称
     */
    public String requestAttribute(String name);

    /**
     * 获取 request Parameter参数
     * @param name 参数名称
     */
    public String requestParameter(String name);

    /**
     * 获取请求上下文
     * @return 返回请求上下文
     */
    public String contextPath() ;

    /**
     * 获取请求上下文绝对路径,会把请求上下文路径(contextPath)拼接在路径之前
     * @param path 路径
     */
    public String url(String path);

    /**
     * 生成cdn Url
     * @param url 资源路径
     */
    public String cdn(String url);

    /**
     * 获取cdn服务器
     */
    public String cdnRoot();

    /**
     * 测试对象，分别返回为true和为false的值
     * 如果是整数类型，1返回true
     * 如果是布尔类型直接判断
     * @param test 测试对象
     * @param trueOut true返回值
     * @param falseOut false返回值
     */
    public String iif(Object test, String trueOut, String falseOut);

    /**
     * 生成span标签label类，成功加label-success类，失败加label-danger类
     * @param test 测试条件
     * @param trueLable true文本
     * @param falseLable false文本
     */
    public String boolLabel(Object test, String trueLable, String falseLable);

    /**
     * 生成span标签label类，成功加label-success类，失败加label-danger类
     * @param test 测试条件
     * @param lable true和false文本相同
     */
    public String boolLabel(Object test, String lable);

    /**
     * 生成span标签label类，成功加label-success类，失败加label-danger类
     * trueLable=是
     * falseLable=否
     * @param test 测试条件
     */
    public String yesnoLabel(Object test);

    /**
     * 生成span标签label类，成功加label-success类，失败加label-danger类
     * trueLable=启用
     * falseLable=禁用
     * @param test 测试条件
     */
    public String statusLabel(Object test);

    /**
     * 测试条件，true返回 readonly，否则返回空字符串
     * @param test 测试对象
     */
    public String isReadonly(Object test);

    /**
     * 测试条件，true返回 selected，否则返回空字符串
     * @param test 测试对象
     */
    public String isSelected(Object test);

    /**
     * 测试条件，true返回 checked，否则返回空字符串
     * @param test 测试对象
     */
    public String isChecked(Object test);

    /**
     * 测试条件，true返回 disabled，否则返回空字符串
     * @param test 测试对象
     */
    public String isDisabled(Object test);

    /**
     * 测试条件，true返回 class="active"，否则返回空字符串
     * @param test 测试对象
     */
    public String isActiveClass(Object test);

    /**
     * 生成状态下拉option
     * 括空标签=全部状态
     * 启用标签=启用
     * 停用标签=禁用
     * 选中值=""
     */
    public String statusSelectOptions();

    /**
     * 生成状态下拉option
     * 括空标签=全部状态
     * 启用标签=启用
     * 停用标签=禁用
     * @param selectedValue 选中值
     */
    public String statusSelectOptions(String selectedValue);

    /**
     * 生成状态下拉option，包括空标签、启用标签、停用标签
     * @param emptyLable 空标签
     * @param enableLable 启用标签
     * @param disableLable 停用标签
     * @param selectedValue 选中值
     */
    public String statusSelectOptions(String emptyLable, String enableLable, String disableLable, String selectedValue);

    /**
     * 根据集合对象生成option选型列表
     * isMultiple=false
     * nameFieldName=name
     * valueFieldName=id
     * spellFieldName=spell
     * @param list 数据集合
     * @param selectedValue 选中值
     */
    public String selectOptions(List list, Object selectedValue);

    /**
     * 根据集合对象生成option选型列表
     * nameFieldName=name
     * valueFieldName=id
     * spellFieldName=spell
     * @param list 数据集合
     * @param selectedValue 选中值
     * @param isMultiple 是否允许多选
     */
    public String selectOptions(List list, Object selectedValue, boolean isMultiple);

    /**
     * 根据集合对象生成option选型列表
     * @param list 数据集合
     * @param selectedValue 选中值
     * @param isMultiple 是否允许多选
     * @param nameFieldName 显示字段名，不能为空
     * @param valueFieldName 值字段名，不能为空
     * @param spellFieldName 简拼字段名，可为空
     */
    public String selectOptions(List list, Object selectedValue, boolean isMultiple,String nameFieldName, String valueFieldName, String spellFieldName);


    /**
     * 根据字典集合生成option选型列表
     * @param list 字典列表对象
     * @param selectedValue 选中值
     */
    public String dicSelectOptions(List<DicItem> list, Object selectedValue);

    /**
     * 根据字典集合生成option选型列表
     * @param list 字典列表对象
     * @param selectedValue 选中值
     * @param isMultiple 是否允许多选
     */
    public String dicSelectOptions(List<DicItem> list, Object selectedValue, boolean isMultiple);
}

@Component
public class SysWebPage extends WebPage {
    /**
     * 用户权限编码鉴权
     * @param code 权限编码
     * @return 拥有返回true
     */
    public boolean auth(String code);

    /**
     * 获取当前登录用户
     */

    /**
     * 根据参数编码获取参数值
     * @param code         参数编码
     * @param defaultValue 找不到参数时返回的默认值
     * @return 返回参数编码对应的参数值
     */
    public String getParam(String code, String defaultValue);

    /**
     * 获取数据字典集合
     * @param code 字典类型编码
     */
    public List<DicItem> getDic(String code);

    /**
     * 根据字典集合生成option选项列表
     * @param code          字典编码
     * @param selectedValue 选中值
     * @param isMultiple    是否允许多选

    public String dicSelectOptions(String code, Object selectedValue, boolean isMultiple);

    /**
     * 根据字典集合生成option选项列表
     * isMultiple=false
     * @param code          字典编码

    /**
     * 根据字典集合生成option选项列表
     * selectedValue=null
     * isMultiple=false
     * @param code 字典编码
     */
    public String dicSelectOptions(String code);
}
